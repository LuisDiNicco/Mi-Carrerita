/**
 * Agrupar materias por semestre
 */
export function groupBySemester<T extends { semester: number }>(
  items: T[],
): Map<number, T[]> {
  return items.reduce((acc, item) => {
    const semester = item.semester;
    if (!acc.has(semester)) {
      acc.set(semester, []);
    }
    acc.get(semester)!.push(item);
    return acc;
  }, new Map<number, T[]>());
}
